#include <Adafruit_Sensor.h>
#include <DHT.h>
#include <WiFi.h>
#include <ThingSpeak.h>

#define DHTPIN 12  
#define DHTTYPE DHT22
#define sensor1 26
#define sensor2 25
DHT dht(DHTPIN, DHTTYPE);

const char* ssid = "Wokwi-GUEST";
const char* password = "";
const char* thingSpeakApiKey = "XL2GSX50KZDISSHI";
const unsigned long channelNumber = 2139037;

WiFiClient client;

void setup() {
  Serial.begin(115200);
  
  dht.begin();
  pinMode(sensor1, OUTPUT);
  pinMode(sensor2, OUTPUT);
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.println("Conectando na rede WiFi...");
  }
  Serial.println("Conectado!");

  ThingSpeak.begin(client);

}

void loop() {
  float hum = dht.readHumidity();
  float temp = dht.readTemperature();
  
  if (isnan(hum) || isnan(temp)) {
    Serial.println("Leitura n찾o identificada!");
    return;
  }
  
  Serial.print("Umidade: ");
  Serial.print(hum);
  Serial.print("%\t");
  Serial.print("Temperatura: ");
  Serial.print(temp);
  Serial.println("째C");

  ThingSpeak.setField(1, temp);
  ThingSpeak.setField(2, hum);
  
  int response = ThingSpeak.writeFields(channelNumber, thingSpeakApiKey);
  if (response == 200) {
    Serial.println("Dados enviados com sucesso!");
  } else {
    Serial.print("Erro ao enviar dados. C처digo de resposta: ");
    Serial.println(response);
  }

  if (temp > 35) {
    digitalWrite(sensor1, HIGH);
    Serial.println("Temperatura ultrapassou 35째C!");
  } else {
    digitalWrite(sensor1, LOW);
  }
  
  if (hum > 70) {
    digitalWrite(sensor2, HIGH);
    Serial.println("Humidade ultrapassou 70%!");
  } else {
    digitalWrite(sensor2, LOW); 
  }
  delay(15000);
}
